;;; config.el -*- lexical-binding: t; -*-
#+end_src

Load all the interested modules. This will tangle all the files and load all the definitions.
#+begin_src emacs-lisp :tangle ./init.el
(unless (file-exists-p "emacs-basic.el")
  (org-babel-tangle-file "emacs-basic.org"))
(load-file "emacs-basic.el")
#+end_src

* GUIX package manifest
#+begin_src emacs-lisp :noweb-ref settings
(setq my/is-guix t)
#+end_src

Generate guido's guix emacs packages manifest, from submodules defintion.
#+begin_src scheme :tangle ./emacs.scm
(add-to-load-path ".")
(use-modules (emacs-basic))
(concatenate-manifests (list emacs-basic-manifest))
#+end_src

* Tangle
Finally we can configure the system, and apply overriding configs.
#+begin_src emacs-lisp :tangle ./init.el :noweb yes
<<settings>>
(emacs-basic/configure)
<<overrides>>
#+end_src
